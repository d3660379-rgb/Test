        // Глобальные переменные для карусели
        let currentCarouselIndex = 0;

        // Создание звездного фона
        function createStars() {
            const container = document.getElementById('stars-container');
            const starsCount = window.innerWidth < 768 ? 80 : 150;
            
            for (let i = 0; i < starsCount; i++) {
                const star = document.createElement('div');
                star.classList.add('star');
                
                const size = Math.random() * 3 + 1;
                const x = Math.random() * 100;
                const y = Math.random() * 100;
                const duration = Math.random() * 5 + 3;
                const delay = Math.random() * 5;
                
                star.style.width = `${size}px`;
                star.style.height = `${size}px`;
                star.style.left = `${x}%`;
                star.style.top = `${y}%`;
                star.style.animationDuration = `${duration}s`;
                star.style.animationDelay = `${delay}s`;
                
                container.appendChild(star);
            }
        }

        // Создание карточек игр для десктопа
        function createDesktopGameCards() {
            const container = document.getElementById('games-desktop-container');
            
            games.forEach((game, index) => {
                const gameCard = document.createElement('div');
                gameCard.className = 'game-card';
                gameCard.setAttribute('data-game-id', game.id);
                gameCard.setAttribute('aria-label', `Игра ${game.name}`);
                
                // Устанавливаем задержку анимации
                gameCard.style.animationDelay = `${0.3 + index * 0.2}s`;
                
                gameCard.innerHTML = `
                    <img src="${game.bgImage}" alt="${game.name}" class="game-bg-image">
                    <div class="game-content">
                        <div class="game-icon">
                            <i data-lucide="${game.icon}"></i>
                        </div>
                        <h3 class="game-title">${game.name}</h3>
                        <p class="game-description">${game.shortDescription}</p>
                        <button class="details-btn" data-game-id="${game.id}">ПОДРОБНЕЕ</button>
                    </div>
                `;
                
                container.appendChild(gameCard);
            });
        }

        // Создание карусели для мобилок
        function createMobileCarousel() {
            const track = document.getElementById('carousel-track');
            const indicatorsContainer = document.getElementById('carousel-indicators');
            
            games.forEach((game, index) => {
                // Создаем карточку для карусели
                const gameCard = document.createElement('div');
                gameCard.className = `game-card-carousel ${index === 0 ? 'active' : ''}`;
                gameCard.setAttribute('data-game-id', game.id);
                gameCard.setAttribute('aria-label', `Игра ${game.name}`);
                
                gameCard.innerHTML = `
                    <img src="${game.bgImage}" alt="${game.name}" class="game-bg-image">
                    <div class="game-content">
                        <div class="game-icon">
                            <i data-lucide="${game.icon}"></i>
                        </div>
                        <h3 class="game-title">${game.name}</h3>
                        <p class="game-description">${game.shortDescription}</p>
                        <button class="details-btn" data-game-id="${game.id}">ПОДРОБНЕЕ</button>
                    </div>
                `;
                
                track.appendChild(gameCard);
                
                // Создаем индикатор
                const indicator = document.createElement('div');
                indicator.className = `indicator ${index === 0 ? 'active' : ''}`;
                indicator.setAttribute('data-index', index);
                indicatorsContainer.appendChild(indicator);
            });
            
            // Настраиваем обработчики для карусели
            setupCarousel();
        }

        // Настройка карусели
        function setupCarousel() {
            const track = document.getElementById('carousel-track');
            const cards = document.querySelectorAll('.game-card-carousel');
            const indicators = document.querySelectorAll('.indicator');
            
            // Обработчик для индикаторов
            indicators.forEach(indicator => {
                indicator.addEventListener('click', function() {
                    const index = parseInt(this.getAttribute('data-index'));
                    scrollToCarouselItem(index);
                });
            });
            
            // Обработчик прокрутки для определения активной карточки
            track.addEventListener('scroll', function() {
                updateActiveCarouselItem();
            });
            
            // Начальная настройка
            updateActiveCarouselItem();
        }

        // Прокрутка к определенному элементу карусели
        function scrollToCarouselItem(index) {
            const track = document.getElementById('carousel-track');
            const cards = document.querySelectorAll('.game-card-carousel');
            
            if (cards[index]) {
                const card = cards[index];
                const cardWidth = card.offsetWidth;
                const gap = 20; // соответствует gap в CSS
                const scrollPosition = index * (cardWidth + gap);
                
                track.scrollTo({
                    left: scrollPosition,
                    behavior: 'smooth'
                });
                
                currentCarouselIndex = index;
                updateCarouselIndicators(index);
            }
        }

        // Обновление активного элемента карусели
        function updateActiveCarouselItem() {
            const track = document.getElementById('carousel-track');
            const cards = document.querySelectorAll('.game-card-carousel');
            
            // Находим ближайшую карточку к центру
            let closestIndex = 0;
            let minDistance = Infinity;
            const trackCenter = track.scrollLeft + track.offsetWidth / 2;
            
            cards.forEach((card, index) => {
                const cardCenter = card.offsetLeft + card.offsetWidth / 2;
                const distance = Math.abs(cardCenter - trackCenter);
                
                if (distance < minDistance) {
                    minDistance = distance;
                    closestIndex = index;
                }
            });
            
            // Обновляем активную карточку
            cards.forEach((card, index) => {
                if (index === closestIndex) {
                    card.classList.add('active');
                } else {
                    card.classList.remove('active');
                }
            });
            
            // Обновляем индикаторы
            updateCarouselIndicators(closestIndex);
            currentCarouselIndex = closestIndex;
        }

        // Обновление индикаторов карусели
        function updateCarouselIndicators(activeIndex) {
            const indicators = document.querySelectorAll('.indicator');
            indicators.forEach((indicator, index) => {
                if (index === activeIndex) {
                    indicator.classList.add('active');
                } else {
                    indicator.classList.remove('active');
                }
            });
        }

        // Открытие модального окна с деталями игры
        function openGameModal(gameId) {
            const game = games.find(g => g.id === gameId);
            if (!game) return;
            
            // Заполняем модальное окно данными
            document.getElementById('modal-game-title').textContent = game.name;
            document.getElementById('modal-description').textContent = game.fullDescription;
            document.getElementById('modal-play-btn').href = game.url;
            document.getElementById('modal-game-info').textContent = 
                `Жанр: ${game.genre} | Сложность: ${game.difficulty} | Время игры: ${game.playTime}`;
            
            // Заполняем скриншоты
            const screenshotsContainer = document.getElementById('modal-screenshots');
            screenshotsContainer.innerHTML = '';
            
            game.screenshots.forEach(screenshot => {
                const screenshotElement = document.createElement('div');
                screenshotElement.className = 'screenshot';
                screenshotElement.innerHTML = `<img src="${screenshot}" alt="Скриншот ${game.name}">`;
                screenshotsContainer.appendChild(screenshotElement);
            });
            
            // Заполняем особенности
            const featuresContainer = document.getElementById('modal-features');
            featuresContainer.innerHTML = '';
            
            game.features.forEach(feature => {
                const featureElement = document.createElement('div');
                featureElement.className = 'feature';
                featureElement.innerHTML = `
                    <div class="feature-title">${feature.title}</div>
                    <div class="feature-description">${feature.description}</div>
                `;
                featuresContainer.appendChild(featureElement);
            });
            
            // Показываем модальное окно
            const modal = document.getElementById('game-modal');
            modal.classList.add('active');
            document.body.style.overflow = 'hidden';
        }

        // Закрытие модального окна
        function closeGameModal() {
            const modal = document.getElementById('game-modal');
            modal.classList.remove('active');
            document.body.style.overflow = 'auto';
        }

        // Обработчики для улучшенной поддержки мобилок
        function setupMobileSupport() {
            // Предотвращаем масштабирование при двойном тапе
            let lastTouchEnd = 0;
            document.addEventListener('touchend', function(event) {
                const now = Date.now();
                if (now - lastTouchEnd <= 300) {
                    event.preventDefault();
                }
                lastTouchEnd = now;
            }, false);
            
            // Обработчики свайпа для карусели
            let touchStartX = 0;
            let touchEndX = 0;
            
            const carousel = document.getElementById('games-carousel-container');
            
            carousel.addEventListener('touchstart', function(e) {
                touchStartX = e.changedTouches[0].screenX;
            }, { passive: true });
            
            carousel.addEventListener('touchend', function(e) {
                touchEndX = e.changedTouches[0].screenX;
                handleSwipe();
            }, { passive: true });
            
            function handleSwipe() {
                const swipeThreshold = 50;
                const diff = touchStartX - touchEndX;
                
                if (Math.abs(diff) > swipeThreshold) {
                    if (diff > 0) {
                        // Свайп влево - следующий
                        scrollToCarouselItem(Math.min(currentCarouselIndex + 1, games.length - 1));
                    } else {
                        // Свайп вправо - предыдущий
                        scrollToCarouselItem(Math.max(currentCarouselIndex - 1, 0));
                    }
                }
            }
        }
